## Problem Summary

- **Current state**: Multiple components fetch `/asociados` independently (Sidebar, Dashboard, etc.); no deduplication or caching.
- **Impact**: Redundant API calls, large payloads, network overhead, UI latency.
- **Finding evidence**: 
  - [Sidebar.tsx:27–35](../app/components/Sidebar.tsx#L27-L35)
  - [Dashboard.tsx:11–20](../app/dashboard/page.tsx#L11-L20)
  - No pagination: data grows linearly; N+1 requests possible on scale.

---

## Solution

Implement **React Query** (TanStack Query) for:
1. **Automatic request deduplication**: Multiple components requesting `/asociados` = single API call.
2. **Built-in caching**: Data cached in memory with configurable TTL.
3. **Pagination**: Backend returns paginated data; frontend manages cursor/offset.
4. **Stale-while-revalidate**: Update cache in background without blocking UI.

### Step 2: Create React Query Provider
**Purpose**: Initialize QueryClientProvider at root level.

**File**: `app/providers.tsx` (create new)

**Content**:
```typescript
"use client";

import { QueryClient, QueryClientProvider } from "@tanstack/react-query";
import { ReactNode } from "react";

const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 1000 * 60 * 5, // 5 minutes
      gcTime: 1000 * 60 * 10, // 10 minutes (formerly cacheTime)
      retry: 1,
      refetchOnWindowFocus: false, // Disable auto-refetch on tab focus to reduce API calls
    },
  },
});

export default function Providers({ children }: { children: ReactNode }) {
  return (
    <QueryClientProvider client={queryClient}>
      {children}
    </QueryClientProvider>
  );
}
```

**Explanation**:
- `staleTime: 5 minutes`: Data considered fresh for 5 min; no refetch within window.
- `gcTime: 10 minutes`: Keep cached data in memory for 10 min even if unused.
- `retry: 1`: Retry once on network failure.
- `refetchOnWindowFocus: false`: Don't refetch when user returns to tab (noisy for public data).

### Step 4: Create Custom Hook for Asociados Query
**Purpose**: Centralize `/asociados` API logic with React Query.

**File**: `hooks/useAsociados.ts` (create new)

**Content**:
```typescript
import { useQuery } from "@tanstack/react-query";
import api from "@/services/api";

export interface Asociado {
  id: number;
  name: string;
  email: string;
  phone?: string;
  created_at: string;
}

export interface PaginatedResponse {
  data: Asociado[];
  pagination: {
    total: number;
    per_page: number;
    current_page: number;
    last_page: number;
  };
}

interface UseAsociadosOptions {
  page?: number;
  perPage?: number;
  enabled?: boolean;
}

/**
 * Fetch asociados with pagination support.
 * Automatically cached and deduplicated by React Query.
 */
export function useAsociados(options: UseAsociadosOptions = {}) {
  const { page = 1, perPage = 50, enabled = true } = options;

  return useQuery({
    queryKey: ["asociados", { page, perPage }], // Unique cache key
    queryFn: async (): Promise<PaginatedResponse> => {
      const response = await api.get("/asociados", {
        params: { page, per_page: perPage },
      });
      return response.data;
    },
    enabled, // Allow conditional fetching
  });
}
```

**Explanation**:
- `queryKey: ["asociados", { page, perPage }]`: Each page gets its own cache entry.
- `queryFn`: Async function that fetches data.
- `enabled`: Allows pausing queries conditionally (e.g., while user not authenticated).

### Step 8: Create Pagination Component (Optional, for Future)
**Purpose**: Reusable pagination UI for paginated lists.

**File**: `app/components/Pagination.tsx` (create new; optional)

**Content**:
```typescript
interface PaginationProps {
  currentPage: number;
  lastPage: number;
  onPageChange: (page: number) => void;
}

export default function Pagination({
  currentPage,
  lastPage,
  onPageChange,
}: PaginationProps) {
  return (
    <div className="flex items-center justify-center gap-2 mt-6">
      <button
        onClick={() => onPageChange(Math.max(1, currentPage - 1))}
        disabled={currentPage === 1}
        className="px-3 py-1 rounded bg-stone-200 disabled:opacity-50"
      >
        Anterior
      </button>
      
      <span>
        Página {currentPage} de {lastPage}
      </span>
      
      <button
        onClick={() => onPageChange(Math.min(lastPage, currentPage + 1))}
        disabled={currentPage === lastPage}
        className="px-3 py-1 rounded bg-stone-200 disabled:opacity-50"
      >
        Siguiente
      </button>
    </div>
  );
}
```

**Usage example** (in a list page):
